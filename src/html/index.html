gulp.task('rubySass', function () {
  gulp.src(paths.scssFiles)
    .pipe($.plumber())
    .pipe($.rubySass({
      r: 'sass-globbing',
      'sourcemap=none': true
    }))
    .pipe($.filter('*.css'))
    .pipe($.pleeease({
      autoprefixer: {
        browsers: ['last 2 versions']
      },
      sourcemaps: true
    }))
    .pipe($.filter('*.css').restore())
    .pipe(gulp.dest(paths.cssDest))
    .pipe(browserSync.reload({ stream: true }));
});
